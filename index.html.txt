    <button onclick="confess()">Confess!</button>
    <div id="confession"></div>

    <script>
        function confess() {
            const name = document.getElementById('crushName').value.trim();
            const msg = document.getElementById('message').value.trim();
            const output = document.getElementById('confession');
            
            if (!name || !msg) {
                output.innerHTML = "<p style='color: red;'>Please fill in both fields!</p>";
                output.style.display = "block";
                return;
            }
            
            output.innerHTML = `<p>Dear ${name},<br>${msg}<br>With love, [Your Name]</p>`;
            output.style.display = "block";
        }
    </script>
